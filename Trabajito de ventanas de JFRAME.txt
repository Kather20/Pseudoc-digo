    int v[] = new int[30];
    int x = 0;
    private void btnGenerarDataActionPerformed(java.awt.event.ActionEvent evt) {                                               
        x = Integer.parseInt(txtNumeroElementos.getText());
        Random rd = new Random();
        for (int i = 0; i < x; i++) {
            v[i] = rd.nextInt(100);

        }
        String vects = "{";
        for (int i = 0; i < x; i++) {
            vects += String.valueOf(v[i]);
            if (i < x - 1) {
                vects += ",";
            }

        }
        vects += "}";
        txtcuadro.setText(vects);

    }                                              

    private void btnrealizarActionPerformed(java.awt.event.ActionEvent evt) {                                            
        int resultado = 0;
        long inicio = 0, fin = 0;
        String tipo = "";
        if (BotonRecursivo.isSelected()) {
            inicio = System.nanoTime();
            tipo = "Recursivo";
            resultado = sumRecursivo(v, x);
            fin = System.nanoTime();
        }
        if (BotonInterativo.isSelected()) {
            tipo="Iterativo";
            inicio = System.nanoTime();
            for (int i = 0; i < x; i++) {
                resultado += v[i];

            }
            fin = System.nanoTime();
        }
        DefaultTableModel vecd=(DefaultTableModel)tabla.getModel();
        Object[] row={x,tipo,inicio,fin,resultado};
        vecd.addRow(row);
    }

    private int sumRecursivo(int[] p, int o) {
        if (o == 0) {
            return p[o];
        } else {
            return p[o] + sumRecursivo(p, o - 1);
        }
    }
